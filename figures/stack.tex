\begin{ocamlcode}
type 'a t = 'a list Atomic.t
let create () = Atomic.make []

let rec push t v =
  let old = Atomic.get t in
  let new_ = v :: old in
  if not (Atomic.compare_and_set
            t old new_)
  then (
    Domain.cpu_relax () ;
    push t v
  )

let rec pop t =
  match Atomic.get t with
  | [] -> None
  | v :: new_ as old ->
      if Atomic.compare_and_set
           t old new_
      then Some v
      else (
        Domain.cpu_relax () ;
        pop t
      )
\end{ocamlcode}
